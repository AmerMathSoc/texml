\subsection{Procedure}

\tracingmacros=1

\begin{algorithmic}[1]
\Procedure{Euclid}{$a,b$}\Comment{The g.c.d. of a and b}
    \State $r\gets a\bmod b$
    \While{$r\not=0$}\Comment{We have the answer if r is 0}
        \State $a\gets b$
        \State $b\gets r$
        \State $r\gets a\bmod b$
    \EndWhile\label{euclidendwhile}
    \State \textbf{return} $b$\Comment{The gcd is b}
\EndProcedure
\end{algorithmic}

\subsection{The \emph{if-then-else} Statement}

% \tracingmacros=1

\begin{algorithmic}[1]
    \If{some condition is true}\Comment{comment}
        \State do some processing
    \ElsIf{some other condition is true}\Comment{comment}
        \State do some different processing
    \ElsIf{some even more bizarre condition is met}\Comment{comment}
        \State do something else
    \Else\Comment{comment}
        \State do the default actions
    \EndIf
\end{algorithmic}

\section{Some longer examples}

\subsection{\emph{if-elsif-else}}

\begin{algorithmic}[3]
    \State $a \leftarrow 1$
    \If{$a$ is even}
        \State \textbf{PRINT} ``$a$ is even''
    \ElsIf{$a$ is odd}
        \State \textbf{PRINT} ``$a$ is odd''
    \Else
        \State \textbf{PRINT} ``$a$ is really weird''
    \EndIf
\end{algorithmic}

\subsection{Nested structures}

\begin{algorithmic}
    \Require $n \geq 0$
    \Ensure $y = x^n$
    \State $y \leftarrow 1$
    \State $X \leftarrow x$
    \State $N \leftarrow n$
    \While{$N \neq 0$}
        \If{$N$ is even}
            \State $X \leftarrow X \times X$
            \State $N \leftarrow N / 2$
        \Else\Comment{$N$ is odd}
            \State $y \leftarrow y \times X$
            \State $N \leftarrow N - 1$
        \EndIf
    \EndWhile
\end{algorithmic}
