    \noAMSmetadata
\fi

\ifx\LoadPackage\undefined \else
    \LoadPackage{Algorithmic}
\fi

\usepackage{algorithmic}

\title{algorithmic}

\parindent0pt

\begin{document}

\maketitle

hello, world

\begin{algorithmic}[1]
\REQUIRE $n \geq 0$
\ENSURE $y = x^n$
\STATE $y \leftarrow 1$
\STATE $X \leftarrow x$
\STATE $N \leftarrow n$
\WHILE{$N \neq 0$}
\IF{$N$ is even}
\STATE $X \leftarrow X \times X$
\STATE $N \leftarrow N / 2$
\ELSE[$N$ is odd]
\STATE $y \leftarrow y \times X$
\STATE $N \leftarrow N - 1$
\ENDIF
\ENDWHILE
\end{algorithmic}

\end{document}
