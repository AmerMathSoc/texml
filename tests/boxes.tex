\documentclass{amsart}

\title{boxes}

\csname noAMSmetadata\endcsname

\csname noTeXMLarticlehistory\endcsname

\parindent0pt

\def\isvoid#1{%
    Box(#1) is
    \ifvoid#1 void (CORRECT)\else not void (INCORRECT)\fi
    \par
}

\def\isnotvoid#1{%
    Box(#1) is
    \ifvoid#1 void (INCORRECT)\else not void (CORRECT)\fi
    \par
}

\def\ishbox#1{%
    Box(#1) is
    \ifhbox#1 an hbox (CORRECT)\else not an hbox (INCORRECT)\fi
    \par
}

\def\isnothbox#1{%
    Box(#1) is
    \ifhbox#1 an hbox (INCORRECT)\else not an hbox (CORRECT)\fi
    \par
}

\def\isvbox#1{%
    Box(#1) is
    \ifvbox#1 a vbox (CORRECT)\else not a vbox (INCORRECT)\fi
    \par
}

\def\isnotvbox#1{%
    Box(#1) is
    \ifvbox#1 a vbox (INCORRECT)\else not a vbox (CORRECT)\fi
    \par
}
